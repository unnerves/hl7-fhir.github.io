7.19.5.1 General-Primary

XML

General Documentation Submission Example (id = "654789")

Raw XML

<SupportingDocumentation xmlns="http://hl7.org/fhir">
  <id value="654789"/>
  <text>
    <status value="generated"/>
    <div xmlns="http://www.w3.org/1999/xhtml"><!-- Snipped for brevity --></div>
  </text>
  <identifier>
    <system value="http://happyvalley.com/elegibility"/>
    <value value="52345"/>
  </identifier>
  <created value="2014-09-21"/>
  <target>
    <reference value="Organization/2"/>
  </target>
  <subject>
    <reference value="Patient/1"/>
  </subject>
  <detail>
    <linkId value="1"/>
    <contentAttachment>
      <contentType value="application/pdf"/>
      <data value="SGVsbG8="/>
      <title value="accident notes 20100201.pdf"/>
    </contentAttachment>
    <dateTime value="2010-02-01"/>
  </detail>
  <detail>
    <linkId value="2"/>
    <contentAttachment>
      <contentType value="application/pdf"/>
      <url value="http://happyvalley.com/docs/AB12345"/>
      <size value="104274"/>
      <hash value="SGVsbG8gdGhlcmU="/>
    </contentAttachment>
    <dateTime value="2010-02-01T10:57:34+01:00"/>
  </detail>
</SupportingDocumentation>
