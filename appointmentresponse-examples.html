5.25.6.1 Example

XML

General Person Example (id = "example")

Raw XML

<AppointmentResponse xmlns="http://hl7.org/fhir">
  <id value="example"/>
  <text>
    <status value="generated"/>
    <div xmlns="http://www.w3.org/1999/xhtml"><!-- Snipped for brevity --></div>
  </text>
  <appointment>
    <reference value="Appointment/example"/>
    <display value="Brian MRI results discussion"/>
  </appointment>
  <individual>
    <reference value="Patient/example"/>
    <display value="Peter James Chalmers"/>
  </individual>
  <participantStatus value="accepted"/>
  <lastModifiedBy>
    <reference value="Practitioner/example"/>
    <display value="Dr Adam Careful"/>
  </lastModifiedBy>
  <lastModified value="2012-09-17"/>
</AppointmentResponse>
